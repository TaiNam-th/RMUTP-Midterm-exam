ตัวอย่างการใช้งาน

ตัวอย่างการใช้งานตัวเลขใน Python พร้อมคำอธิบายและผลลัพธ์

Integer (จำนวนเต็ม)

                        
                            # การใช้งาน Integer
                            positive = 42        # จำนวนเต็มบวก
                            negative = -17       # จำนวนเต็มลบ
                            zero = 0            # ศูนย์
                            big_number = 1_000_000  # ใช้ _ เพื่อให้อ่านง่าย

                            # แสดงผลและประเภทข้อมูล
                            print(f"positive: {positive} - type: {type(positive)}")
                            print(f"negative: {negative} - type: {type(negative)}")
                            print(f"big number: {big_number:,}")  # แสดงผลแบบมีตัวคั่นหลักพัน

Float (จำนวนทศนิยม)

                        
                            # การใช้งาน Float
                            pi = 3.14159
                            small = 0.0001
                            scientific = 2.5e-3  # รูปแบบ scientific notation
                            infinite = float('inf')  # ค่าอนันต์

                            # แสดงผลในรูปแบบต่างๆ
                            print(f"pi: {pi:.2f}")  # แสดง 2 ตำแหน่งทศนิยม
                            print(f"scientific: {scientific:e}")  # แสดงในรูปแบบ scientific
                            print(f"infinite: {infinite}")

Complex Numbers (จำนวนเชิงซ้อน)

                        
                            # การใช้งาน Complex Numbers
                            z1 = 2 + 3j          # สร้างโดยตรง
                            z2 = complex(1, 2)   # สร้างผ่าน constructor

                            # การเข้าถึงส่วนประกอบ
                            print(f"z1 = {z1}")
                            print(f"Real part: {z1.real}")
                            print(f"Imaginary part: {z1.imag}")
                            print(f"z2 in different format: {z2.real} + {z2.imag}j")
